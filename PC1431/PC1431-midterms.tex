  \section{Position, velocity, \& accel of a particle in 1D}
  \subsection{$x$, $v_x$, $a_x$}
  \begin{itemize}
    \item $x = x(t)$
    \item $v_x(t) = \frac{d}{dt} x(t)$
    \item $a_x(t) = \frac{d}{dt}v_x(t) = \frac{d^2}{dt^2}x(t)$
  \end{itemize}
  \subsection{A particle at rest at $x_0$}
  $x(t) = x_0$, $v_x = 0$, $a_x = 0$
  \subsection{A particle moving with constant velocity $v_{0x}$}
  $x(t) = x_0 + v_{0x}t$, $v_x = v_{0x}$, $a_x = 0$
  \subsection{A particle moving with constant acceleration $a_{0x}$}
  \begin{itemize}
    \item $x(t) = x_0 + v_{0x}t + \frac12a_{0x}t^2$
    \item $v_x = v_{0x} + a_{0x}t$
    \item $a_x = a_{0x}$
    \item Freely-falling, $a_{0y} = -g$, $g = 9.80\ m/s^2$
    \item $2a_{0x}(x - x_0) = v_x^2 - v_{0x}^2$ AKA $2as = v^2 - u^2$
  \end{itemize}
  \subsection{Sinusoidal functions of time}
  \begin{itemize}
    \item $x(t) = r_0\cos(\theta_0 + \omega_0t)$ AND $x(0) = r_0\cos\theta_0$
    \item $v_x = -\omega_0r_0\sin(\theta_0 + \omega_0t)$ AND $v_x(0) = -r_0\omega_0\sin\theta_0$
    \item $a_x = -\omega_0^2r_0\cos(\theta_0 + \omega_0t) = \omega_0^2x(t)$
  \end{itemize}
  \subsection{Exponential function of time}
  \begin{itemize}
    \item $y(t) = y_0 - \frac{g}{\gamma}[1-\exp(-\gamma t)] + \frac{g}{\gamma^2}[1 - \exp(-\gamma t)]$
    \item $y(0) = y_0$
    \item $v_y(t) = -\frac{g}{\gamma} + v_{0y}\exp(-\gamma t) + \frac{g}{\gamma}\exp(-\gamma t)$
    \item $v_y(0) = v_{0y}$ AND as $t \to \infty$, $v_y \to -\frac{g}{\gamma} = v_{terminal}$
    \item $a_y(t) = -\gamma v_{0y}\exp(-\gamma t) -g\exp(-\gamma t)$
    \item $a_y(0) = -\gamma v_{0y} - g$ AND as $t\to\infty$, $a_y\to 0$
    \item $a_y(t) = -\gamma v_y(t) - g$
  \end{itemize}
  \subsection{$a_x$, $v_x$, $x$}
  \begin{itemize}
    \item $v_x(t) = v_{0x} + \int_0^ta_x(t')dt'$
    \item $x(t) = x_0 + \int_0^tv_x(t')dt'$
  \end{itemize}

  \section{Position, velocity, \& accel of a particle in 2D}
  \begin{itemize}
    \item $x = x(t), y = y(t)$
    \item $v = \sqrt{v_x^2 + v_y^2}$, $a = \sqrt{a_x^2 + a_y^2}$
  \end{itemize}
  \subsection{Projectile motion}
  \begin{itemize}
    \item $y(t) = -\frac{g}{2v_{0x}^2}(x - x_0) + \frac{v_{0y}}{v_{0x}}(x - x_0) + y_0$
  \end{itemize}
\end{multicols*}
